dpll (function)
  if the function is empty
    return SAT
  if the function contains any zeros
    return unSAT

  while the function contains some unit clauses or pure literals
    if there is a unit clause
      simplify the function and move to the next iteration of the loop
    otherwise
      move to pure literal simplification
  end 

  Make an assignment based on the term with the fewest literals
  
  if dpll(function with assignment) returns SAT
    return SAT
  otherwise
    change the assignment
    return dpll(function with changed assignment)
end
